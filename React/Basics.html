Using Button and OnClick

Note:We cannot add style property to default button. But note that we can create custom button. 
            
import { useState } from 'react';
import { StyleSheet, Text, View ,Button } from 'react-native';


export default function App() {

    const [name,setName] = useState('shaun');

    const clickHandler = () => {
        setName('Diwas');
        }


    return (
    <View style={styles.container}>
        <Text>My Name is {name}</Text>
        <View style={styles.buttonContainer}>
        <Button title='Update State'  onPress={clickHandler}/>
        </View>
    </View>
    );
}

const styles = StyleSheet.create({
    container: {
    flex: 1,
    backgroundColor: 'red',
    alignItems: 'center',
    justifyContent: 'center',
    },
    buttonContainer:{
    marginTop:5
    }
});

Change Object on Button Click 

            
            
import { useState } from 'react';
import { StyleSheet, Text, View ,Button } from 'react-native';


export default function App() {

  const [name,setName] = useState('shaun');
  
    const [person,setPerson] = useState({name:'mario',age:40});


  const clickHandler = () => {
      setName('Diwas');
      setPerson({name:'Diwas',age:78});
  }

  return (
    <View style={styles.container}>
        <Text>His name is {person.name} and his is {person.age} years old.</Text>
        <Text>My Name is {name}</Text>
    <View style={styles.buttonContainer}>
        <Button title='Update State'  onPress={clickHandler}/>
        </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: 'red',
    alignItems: 'center',
    justifyContent: 'center',
  },
  buttonContainer:{
    marginTop:5
  }
});

Flat View

  • For large list FlatList is better because it lazy load the list. And when you scroll , it load remaining from the list.
  • You donot have to wrap it with ScrollView for scrolling . It has auto scrolling effect
  • You donot have to use Key Property.

OnClick on Text Component, Not on Button Component

TouchableOpacity : A wrapper for making views respond properly to touches. On key down, the opacity of the wrapped view is decreased, dimming it. (source) 
            
import { useState } from 'react';
import { StyleSheet, Text, View, FlatList, TouchableOpacity } from 'react-native';

export default function App() {
    const [people, setPeople] = useState([
    { name: 'shaun', id: '1' },
    { name: 'yoshi', id: '2' },
    { name: 'mario', id: '3' },
    { name: 'luigi', id: '4' },
    { name: 'peach', id: '5' },
    { name: 'toad', id: '6' },
    { name: 'bowser', id: '7' },
    ]);

    const pressHandler = (id) => {
    console.log(id);
    };

    return (
    <View style={styles.container}>

        <FlatList 
        numColumns={2}
        keyExtractor={(item) => item.id} 
        data={people} 
        renderItem={({ item }) => (
        <TouchableOpacity onPress={() => pressHandler(item.id)}>
            <Text style={styles.lstStyle}>{item.name}</Text>
            </TouchableOpacity>
        )}
        />

    </View>
    );
}

const styles = StyleSheet.create({
    container: {
    flex: 1,
